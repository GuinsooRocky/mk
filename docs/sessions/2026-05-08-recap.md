## 六、启动 / Rebuild 命令

```bash
# 启动（如果 MK 挂掉了）
open /Users/lengmo/Desktop/my-code/mike/client/.build/MK.app

# 改了源码后 rebuild + 重启
cd /Users/lengmo/Desktop/my-code/mike/client
swift build && \
  cp .build/debug/MK .build/MK.app/Contents/MacOS/MK && \
  codesign --force --sign "MK Development" .build/MK.app && \
  killall MK; open .build/MK.app

# 改了图标 PNG 后重新装 .icns
swift scripts/make-icon.swift /tmp/mk_icon.png
PNG=/tmp/mk_icon.png; ICONSET=/tmp/MK.iconset
rm -rf $ICONSET && mkdir -p $ICONSET
for sz in 16 32 128 256 512; do
  sips -z $sz $sz $PNG --out $ICONSET/icon_${sz}x${sz}.png > /dev/null 2>&1
  sips -z $((sz*2)) $((sz*2)) $PNG --out $ICONSET/icon_${sz}x${sz}@2x.png > /dev/null 2>&1
done
cp $PNG $ICONSET/icon_512x512@2x.png
iconutil -c icns -o /tmp/MK.icns $ICONSET
cp /tmp/MK.icns .build/MK.app/Contents/Resources/AppIcon.icns
touch .build/MK.app
codesign --force --sign "MK Development" .build/MK.app
killall Finder
```
